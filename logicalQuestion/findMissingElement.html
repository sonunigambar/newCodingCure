Write a program to find the missing element of a given Array?

Input1: arr[]=1,3,5,7,9

Output: 2,4,6,8

     import java.util.ArrayList;
import java.util.List;

public class MissingElementsFinder {

    public static void main(String[] args) {
        int[] arr = {1, 3, 5, 7, 9};  // Input array
        List missingElements = findMissingElements(arr, 2);  // Difference is 2

        System.out.println("Missing elements: " + missingElements);
    }

    public static List findMissingElements(int[] arr, int difference) {
        List missingElements = new ArrayList<>();

        for (int i = 1; i < arr.length; i++) {
            int current = arr[i - 1];
            int next = arr[i];
            
            // Calculate the expected value in between if there's a gap
            while (current + difference < next) {
                current += difference;
                missingElements.add(current);
            }
        }

        return missingElements;
    }
}
